CREATE TABLE VACANCIESS(
VACANCYID INT PRIMARY KEY,
VACANCYNAME VARCHAR2(50) NOT NULL
);
CREATE TABLE OTRASLS(
OTRASLID INT PRIMARY KEY,
OTRASLNAME VARCHAR2(50) NOT NULL
);
CREATE TABLE FIRMS(
FIRMAID INT PRIMARY KEY,
FIRMANAME VARCHAR2(50) NOT NULL,
OTRASL INT REFERENCES OTRASLS(OTRASLID) NOT NULL,
SHTAB_FLAT VARCHAR2(50)
);
CREATE TABLE REQUIREMENTS(
REQUIREMENTID INT PRIMARY KEY,
REQUIREMENTNAME VARCHAR2(50)
);
CREATE TABLE SPECIALIZATION(
SPECID INT PRIMARY KEY,
SPECNAME VARCHAR2(50) NOT NULL
);
CREATE TABLE SPEC_FIRM(
FIRMAID INT REFERENCES FIRMS(FIRMAID),
SPECID INT REFERENCES SPECIALIZATION(SPECID) NOT NULL
);
CREATE TABLE REQUIR_VACAN(
VACANCYID INT REFERENCES VACANCIESS(VACANCYID),
REQUIREMENTID INT REFERENCES REQUIREMENTS(REQUIREMENTID) NOT NULL
);
CREATE TABLE TYPEWORK(
WORKID INT PRIMARY KEY,
TYPE VARCHAR2(50) NOT NULL
);

CREATE TABLE OFFERS(
VACANCYID INT REFERENCES VACANCIESS(VACANCYID)NOT NULL,
FIRMAID INT REFERENCES FIRMS(FIRMAID) NOT NULL,
WORKID INT REFERENCES TYPEWORK(WORKID) NOT NULL,
DOLZHNOST VARCHAR2(50) NOT NULL,
DATE_LOC DATE
);
CREATE TABLE SKILLS(
SKILLID INT PRIMARY KEY,
SKILLNAME VARCHAR2(50)
);

CREATE TABLE SOISKATELS(
SOISKATELID INT PRIMARY KEY,
SOISKATELNAME VARCHAR2(50) NOT NULL,
TELEPHONE VARCHAR2(50)
);
CREATE TABLE SKILLS_SOISK(
SOISKATELID INT REFERENCES SOISKATELS(SOISKATELID) NOT NULL,
SKILLID INT REFERENCES SKILLS(SKILLID) NOT NULL
);
CREATE TABLE ZAYAVKA(
SOISKATELID INT REFERENCES SOISKATELS(SOISKATELID),
VACANCYID INT REFERENCES VACANCIESS(VACANCYID)
);
